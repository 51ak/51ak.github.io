Oracle增加表审计

增加表审计

# 查看现在有哪些审计对象
select * from dba_obj_audit_opts;
# 开启审计
audit select,update,insert,delete on 用户.表名 by access
# 关闭审计
noaudit select,update,insert,delete on 用户.表名
# 查看审计结果
select * from dba_audit_trail

清空审计

sqlplus / as sysdba


truncate table sys.aud$;
